126:     /**
127:      * Creates a pkpass file
128:      *
129:      * @param  Passbook\PassInterface $pass
130:      * @return SplFileObject
131:      */
132:     public function package(PassInterface $pass)
133:     {
134:         // Serialize pass
135:         $pass->setPassTypeIdentifier($this->passTypeIdentifier);
136:         $pass->setTeamIdentifier($this->teamIdentifier);
137:         $pass->setOrganizationName($this->organizationName);
138:         $json = SerializerBuilder::create()->build()->serialize($pass, 'json');
139: 
140:         $outputPath = rtrim($this->getOutputPath(), DIRECTORY_SEPARATOR) . DIRECTORY_SEPARATOR;
141:         $passDir = $outputPath . $pass->getSerialNumber() . DIRECTORY_SEPARATOR;
142:         $passDirExists = file_exists($passDir);
143:         if ($passDirExists && !$this->override) {
144:             throw new FileException("Temporary pass directory already exists");
145:         } elseif (!$passDirExists && !mkdir($passDir, 0777, true)) {
146:             throw new FileException("Couldn't create temporary pass directory");
147:         }
148: 
149:         // Pass.json
150:         $passJSONFile = $passDir . 'pass.json';
151:         file_put_contents($passJSONFile, $json);
152: 
153:         // Images
154:         foreach ($pass->getImages() as $image) {
155:             $fileName = $passDir . $image->getContext();
156:             if ($image->getIsRetina()) $fileName .= '@2x';
157:             $fileName .= '.' . $image->getExtension();
158:             copy($image->getRealPath(), $fileName);
159:         }
160: 
161:         // Manifest.json
162:         $manifestJSONFile = $passDir . 'manifest.json';
163:         $manifest = array();
164:         foreach (scandir($passDir) as $file)
165:         {
166:             if ($file == '.' or $file == '..') continue;
167:             $manifest[$file] = sha1_file($passDir . $file);
168:         }
169:         file_put_contents($manifestJSONFile, json_encode($manifest));
170: 
171:         // Signature
172:         $signatureFile = $passDir . 'signature';
173:         $p12 = file_get_contents($this->p12->getRealPath());
174:         $certs = array();
175:         if (openssl_pkcs12_read($p12, $certs, $this->p12->getPassword()) == true) {
176:             $certdata = openssl_x509_read($certs['cert']);
177:             $privkey = openssl_pkey_get_private($certs['pkey'], $this->p12->getPassword());
178:             openssl_pkcs7_sign($manifestJSONFile, $signatureFile, $certdata, $privkey, array(), PKCS7_BINARY | PKCS7_DETACHED, $this->wwdr->getRealPath());
179:             // Get signature content
180:             $signature = @file_get_contents($signatureFile);
181:             // Check signature content
182:             if (!$signature) throw new FileException("Couldn't read signature file.");
183:             // Delimeters
184:             $begin = 'filename="smime.p7s"' . PHP_EOL . PHP_EOL;
185:             $end = PHP_EOL . PHP_EOL . '------';
186:             // Convert signature
187:             $signature = substr($signature, strpos($signature, $begin) + strlen($begin));
188:             $signature = substr($signature, 0, strpos($signature, $end));
189:             $signature = base64_decode($signature);
190:             // Put new signature
191:             if (!file_put_contents($signatureFile, $signature)) throw new FileException("Couldn't write signature file.");
192:         } else {
193:             throw new FileException("Error reading certificate file");
194:         }
195: 
196:         // Zip pass
197:         $zipFile = $outputPath . $pass->getSerialNumber() . self::PASS_EXTENSION;
198:         $zip = new ZipArchive();
199:         if (!$zip->open($zipFile, $this->override ? ZIPARCHIVE::OVERWRITE : ZipArchive::CREATE)) throw new FileException("Couldn't open zip file.");

201:             $zip->addFile($passDir);
202:             while (false !== ($entry = readdir($handle))) {
203:                 if ($entry == '.' or $entry == '..') continue;
204:                 $zip->addFile($passDir . $entry, $entry);
205:             }
206:             closedir($handle);
207:         } else {
208:             throw new FileException("Error reading pass directory");
209:         }
210:         $zip->close();

212:         $this->rrmdir($passDir);
213:         return new SplFileObject($zipFile);
214:     }
